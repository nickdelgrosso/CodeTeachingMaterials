freq_min, freq_max = 0, 10000
n_fft = 2048 * 128

file_list = [name for name in os.listdir(os.getcwd()) if '.wav' in name]
ref_file = filter( lambda x: 'Fan0_' in x, file_list)[0]
ref_file_idx = file_list.index(ref_file)
display(list(enumerate(file_list)))

print('Reference Sound File is at position {0}: {1}'.format(ref_file_idx, ref_file))

power_list = []
for file_name in file_list:
    
    # Read Data
    samp_rate, data = wavfile.read(file_name)
    data = data.astype(float) / 2**16  # Rescale to percentage of max mic amplitude.
    
    # Calculate Periodogram
    freq, power = periodogram(data, samp_rate, nfft=n_fft)

    # Frequency Cutoff
    power, freq = power[freq < freq_max], freq[freq < freq_max]
    
    power_list.append(power)
    
# Plot an example periodogram (in this case, the last one.)
idx = 6
plt.plot(freq, power_list[idx])
plt.title(file_list[idx])
plt.xlabel('Frequency (Hz)')
plt.ylabel('Amplitute (% of Max Amplitude)')

def total_energy(power):
    """Gets some single number out of an array of power values across freqs."""
    tot = np.sum(power)
    #tot = np.log10(tot)
    return tot
    
ref_energy = total_energy(power_list[ref_file_idx])
tot_energy, rel_energy = [], []
for power in power_list:
    temp = total_energy(power)
    
    tot_energy.append(temp) 
    rel_energy.append( temp / ref_energy)
    
# Plot Total Energy in Each File    
plt.figure()    
plt.plot(tot_energy)
plt.xlabel('File Number')
plt.ylabel('Total Amplitude, Across Freqs')

# Plot Relative Energy to Reference File for Each File
plt.figure()
plt.plot(rel_energy)
plt.xlabel('File Number')
plt.ylabel('Relative Amplitude (x1 / x0)')
